1. SELECT COUNT(id) AS totalNumber_emp
   FROM employees;

2. SELECT SUM(salary) AS total_salary,
          department
   FROM employees
   WHERE department = 'IT'
   GROUP BY department;

3. SELECT AVG(salary) AS avg_salary 
   FROM employees 
   WHERE department = 'HR'
   GROUP BY department;

4. SELECT MIN(salary) AS lowest_salary,
          MAX(salary) AS highest_salary
   FROM employees;
5. SELECT SUM(salary) AS total_salary,
          department
   FROM employees
   GROUP BY department;
6. SELECT COUNT(id) AS number_employees,
          city
   FROM employees
   GROUP BY city;

7. SSELECT AVG(salary) AS avg_salary,
           department 
   FROM employees
   GROUP BY department
   ORDER BY AVG(salary) AS avg_salary DESC;
